## args, kwargs

``` python
def print_all_args(*args, **kwargs):
    for arg in args:
        print (arg)
    for key, value in kwargs.items():
        print (key, value)

def wrapper(*args):
    print_a_and_b(*args)
    
def print_a_and_b(a, b):
    print (a, b)

wrapper(1, 2)
# => 1 2

some_dict = {'val1':4, 'val2':8}
print_all_args(1, 2, foo='bar', toto='titi', *some_dict)
# => 
# 1
# 2
# val1
# val2
# foo bar
# toto titi
```
